
program => statement* EOF

statement => expressionStatement | printStatement | varDeclearationStatement

expressionStatement => expression ";"
printStatement => "print " expression ";"  //temporary

//working in progress
varDeclearationStatement => varType variable_name
varType => "int"

expression => binary | unary | literal

binary  => expression operator expression
unary   => "-" unary | primary
primary => NUMBER | STRING | IDENTIFIER | "true" | "false"
variable_name => IDENTIFIER

operator => "+" | "-" | "*" | "/" | "==" | "!=" | "&&" | "||"
